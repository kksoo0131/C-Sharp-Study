``` bash
  
# 현재 작업중인 디렉터리를 보여준다.
pwd

# 현재 디렉터리의 파일과 디렉터리를 보여준다.
ls

# 현재 디렉터리를 이동한다.
cd [디렉터리 이름]

# 디렉터리를 생성한다.
mkdir [디렉터리 이름]

# 파일을 복사한다.
cp [복사할 파일 경로] [새로운 파일 경로]

# 파일의 이름을 변경하거나 이동한다.
mv [현재 파일 경로/이름] [새로운 파일 경로/이름]

# 파일을 삭제한다.
rm [파일 경로/이름]

# 파일의 내용을 확인한다.
cat [파일 이름]

# 빈 파일을 생성한다.
touch [파일 이름]

# 화면에 글자를 표시한다.
echo [표시할 글자]

# 아이피 정보를 확인한다.
ip addr

# 아이피 정보를 확인한다.
ifconfig

# 네트워크 상태를 확인한다.
ss

# 서버의 포트를 확인한다.
nc

# 어떤 명령어가 어디있는지 확인할 때 사용한다.
which [명령어 이름]

# 파일의 마지막 부분을 보여준다.
tail [파일 이름]

# 파일이나 디렉터리를 찾아준다.
find [검색할 디렉터리] -name [파일/디렉터리 이름]

# 프로세스 상태를 보여준다.
ps

# 파일에서 패턴을 만족하는 문자열이 있는지 확인한다.
grep [패턴] [파일 이름]

# 프로세스를 종료시킨다.
kill [프로세스 ID]

# 리눅스에서 사용하는 텍스트 에디터 중 하나이다.
vim [파일 이름]
